40def cleanse_setting(key, value): 

41 """Cleanse an individual setting key/value of sensitive content. 

42 

43 If the value is a dictionary, recursively cleanse the keys in 

44 that dictionary. 

45 """ 

46 try: 

47 if HIDDEN_SETTINGS.search(key): 

48 cleansed = CLEANSED_SUBSTITUTE 

49 else: 

50 if isinstance(value, dict): 

51 cleansed = {k: cleanse_setting(k, v) for k, v in value.items()} 

52 else: 

53 cleansed = value 

54 except TypeError: 

55 # If the key isn't regex-able, just return as-is. 

56 cleansed = value 

57 

58 if callable(cleansed): 

59 # For fixing #21345 and #23070 

60 cleansed = CallableSettingWrapper(cleansed) 

61 

62 return cleansed

144 def get_post_parameters(self, request): 

145 """ 

146 Replaces the values of POST parameters marked as sensitive with 

147 stars (*********). 

148 """ 

149 if request is None:

169 def cleanse_special_types(self, request, value): 

170 try: 

171 # If value is lazy or a complex object of another kind, this check 

172 # might raise an exception. isinstance checks that lazy 

173 # MultiValueDicts will have a return value. 

174 is_multivalue_dict = isinstance(value, MultiValueDict) 

175 except Exception as e: 

176 return '{!r} while evaluating {!r}'.format(e, value) 

177 

178 if is_multivalue_dict: 

179 # Cleanse MultiValueDicts (request.POST is the one we usually care about) 

180 value = self.get_cleansed_multivaluedict(request, value) 

181 return value

183 def get_traceback_frame_variables(self, request, tb_frame): 

184 """ 

185 Replaces the values of variables marked as sensitive with 

186 stars (*********). 

187 """ 

188 # Loop through the frame's callers to see if the sensitive_variables 

189 # decorator was used. 

190 current_frame = tb_frame.f_back 

191 sensitive_variables = None 

192 while current_frame is not None: 

193 if (current_frame.f_code.co_name == 'sensitive_variables_wrapper' and 

194 'sensitive_variables_wrapper' in current_frame.f_locals): 

195 # The sensitive_variables decorator was used, so we take note 

196 # of the sensitive variables' names. 

197 wrapper = current_frame.f_locals['sensitive_variables_wrapper'] 

198 sensitive_variables = getattr(wrapper, 'sensitive_variables', None) 

199 break 

200 current_frame = current_frame.f_back 

201 

202 cleansed = {} 

203 if self.is_active(request) and sensitive_variables: 

204 if sensitive_variables == '__ALL__': 

205 # Cleanse all variables 

206 for name, value in tb_frame.f_locals.items(): 

207 cleansed[name] = CLEANSED_SUBSTITUTE 

208 else: 

209 # Cleanse specified variables 

210 for name, value in tb_frame.f_locals.items(): 

211 if name in sensitive_variables: 

212 value = CLEANSED_SUBSTITUTE 

213 else: 

214 value = self.cleanse_special_types(request, value) 

215 cleansed[name] = value 

216 else: 

217 # Potentially cleanse the request and any MultiValueDicts if they 

218 # are one of the frame variables. 

219 for name, value in tb_frame.f_locals.items(): 

220 cleansed[name] = self.cleanse_special_types(request, value) 

221 

222 if (tb_frame.f_code.co_name == 'sensitive_variables_wrapper' and 

223 'sensitive_variables_wrapper' in tb_frame.f_locals): 

224 # For good measure, obfuscate the decorated function's arguments in 

225 # the sensitive_variables decorator's frame, in case the variables 

226 # associated with those arguments were meant to be obfuscated from 

227 # the decorated function's frame. 

228 cleansed['func_args'] = CLEANSED_SUBSTITUTE 

229 cleansed['func_kwargs'] = CLEANSED_SUBSTITUTE 

230 

231 return cleansed.items()

336 def _get_lines_from_file(self, filename, lineno, context_lines, loader=None, module_name=None): 

337 """ 

338 Returns context_lines before and after lineno from file. 

339 Returns (pre_context_lineno, pre_context, context_line, post_context). 

340 """ 

341 source = None 

342 if loader is not None and hasattr(loader, "get_source"): 

343 try: 

344 source = loader.get_source(module_name) 

345 except ImportError: 

346 pass 

347 if source is not None: 

348 source = source.splitlines() 

349 if source is None: 

350 try: 

351 with open(filename, 'rb') as fp: 

352 source = fp.read().splitlines() 

353 except (OSError, IOError): 

354 pass 

355 if source is None: 

356 return None, [], None, [] 

357 

358 # If we just read the source from a file, or if the loader did not 

359 # apply tokenize.detect_encoding to decode the source into a Unicode 

360 # string, then we should do that ourselves. 

361 if isinstance(source[0], six.binary_type): 

362 encoding = 'ascii' 

363 for line in source[:2]: 

364 # File coding may be specified. Match pattern from PEP-263 

365 # (http://www.python.org/dev/peps/pep-0263/) 

366 match = re.search(br'coding[:=]\s*([-\w.]+)', line) 

367 if match: 

368 encoding = match.group(1).decode('ascii') 

369 break 

370 source = [six.text_type(sline, encoding, 'replace') for sline in source] 

371 

372 lower_bound = max(0, lineno - context_lines) 

373 upper_bound = lineno + context_lines 

374 

375 pre_context = source[lower_bound:lineno] 

376 context_line = source[lineno] 

377 post_context = source[lineno + 1:upper_bound] 

378 

379 return lower_bound, pre_context, context_line, post_context

381 def get_traceback_frames(self): 

382 def explicit_or_implicit_cause(exc_value): 

383 explicit = getattr(exc_value, '__cause__', None) 

384 implicit = getattr(exc_value, '__context__', None) 

385 return explicit or implicit 

386 

387 # Get the exception and all its causes 

388 exceptions = [] 

389 exc_value = self.exc_value 

390 while exc_value: 

391 exceptions.append(exc_value) 

392 exc_value = explicit_or_implicit_cause(exc_value) 

393 

394 frames = [] 

395 # No exceptions were supplied to ExceptionReporter 

396 if not exceptions: 

397 return frames 

398 

399 # In case there's just one exception (always in Python 2, 

400 # sometimes in Python 3), take the traceback from self.tb (Python 2 

401 # doesn't have a __traceback__ attribute on Exception) 

402 exc_value = exceptions.pop() 

403 tb = self.tb if six.PY2 or not exceptions else exc_value.__traceback__ 

404 

405 while tb is not None: 

406 # Support for __traceback_hide__ which is used by a few libraries 

407 # to hide internal frames. 

408 if tb.tb_frame.f_locals.get('__traceback_hide__'): 

409 tb = tb.tb_next 

410 continue 

411 filename = tb.tb_frame.f_code.co_filename 

412 function = tb.tb_frame.f_code.co_name 

413 lineno = tb.tb_lineno - 1 

414 loader = tb.tb_frame.f_globals.get('__loader__') 

415 module_name = tb.tb_frame.f_globals.get('__name__') or '' 

416 pre_context_lineno, pre_context, context_line, post_context = self._get_lines_from_file( 

417 filename, lineno, 7, loader, module_name, 

418 ) 

419 if pre_context_lineno is not None: 

420 frames.append({ 

421 'exc_cause': explicit_or_implicit_cause(exc_value), 

422 'exc_cause_explicit': getattr(exc_value, '__cause__', True), 

423 'tb': tb, 

424 'type': 'django' if module_name.startswith('django.') else 'user', 

425 'filename': filename, 

426 'function': function, 

427 'lineno': lineno + 1, 

428 'vars': self.filter.get_traceback_frame_variables(self.request, tb.tb_frame), 

429 'id': id(tb), 

430 'pre_context': pre_context, 

431 'context_line': context_line, 

432 'post_context': post_context, 

433 'pre_context_lineno': pre_context_lineno + 1, 

434 }) 

435 

436 # If the traceback for current exception is consumed, try the 

437 # other exception. 

438 if six.PY2: 

439 tb = tb.tb_next 

440 elif not tb.tb_next and exceptions: 

441 exc_value = exceptions.pop() 

442 tb = exc_value.__traceback__ 

443 else: 

444 tb = tb.tb_next 

445 

446 return frames

461def technical_404_response(request, exception): 

462 "Create a technical 404 error response. The exception should be the Http404." 

463 try: 

464 error_url = exception.args[0]['path'] 

465 except (IndexError, TypeError, KeyError): 

466 error_url = request.path_info[1:] # Trim leading slash 

467 

468 try: 

469 tried = exception.args[0]['tried'] 

470 except (IndexError, TypeError, KeyError): 

471 tried = [] 

472 else: 

473 if (not tried or ( # empty URLconf 

474 request.path == '/' and 

475 len(tried) == 1 and # default URLconf 

476 len(tried[0]) == 1 and 

477 getattr(tried[0][0], 'app_name', '') == getattr(tried[0][0], 'namespace', '') == 'admin' 

478 )): 

479 return default_urlconf(request) 

480 

481 urlconf = getattr(request, 'urlconf', settings.ROOT_URLCONF) 

482 if isinstance(urlconf, types.ModuleType): 

483 urlconf = urlconf.__name__ 

484 

485 caller = '' 

486 try: 

487 resolver_match = resolve(request.path) 

488 except Resolver404: 

489 pass 

490 else: 

491 obj = resolver_match.func 

492 

493 if hasattr(obj, '__name__'): 

494 caller = obj.__name__ 

495 elif hasattr(obj, '__class__') and hasattr(obj.__class__, '__name__'): 

496 caller = obj.__class__.__name__ 

497 

498 if hasattr(obj, '__module__'): 

499 module = obj.__module__ 

500 caller = '%s.%s' % (module, caller) 

501 

502 t = DEBUG_ENGINE.from_string(TECHNICAL_404_TEMPLATE) 

503 c = Context({ 

504 'urlconf': urlconf, 

505 'root_urlconf': settings.ROOT_URLCONF, 

506 'request_path': error_url, 

507 'urlpatterns': tried, 

508 'reason': force_bytes(exception, errors='replace'), 

509 'request': request, 

510 'settings': get_safe_settings(), 

511 'raising_view_name': caller, 

512 }) 

513 return HttpResponseNotFound(t.render(c), content_type='text/html')
