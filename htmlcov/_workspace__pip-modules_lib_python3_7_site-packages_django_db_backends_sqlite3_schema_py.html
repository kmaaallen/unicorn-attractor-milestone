18 def __enter__(self): 

19 with self.connection.cursor() as c: 

20 # Some SQLite schema alterations need foreign key constraints to be 

21 # disabled. This is the default in SQLite but can be changed with a 

22 # build flag and might change in future, so can't be relied upon. 

23 # We enforce it here for the duration of the transaction. 

24 c.execute('PRAGMA foreign_keys') 

25 self._initial_pragma_fk = c.fetchone()[0] 

26 c.execute('PRAGMA foreign_keys = 0') 

27 return super(DatabaseSchemaEditor, self).__enter__()

32 # Restore initial FK setting - PRAGMA values can't be parametrized

35 def quote_value(self, value): 

36 # The backend "mostly works" without this function and there are use 

37 # cases for compiling Python without the sqlite3 libraries (e.g. 

38 # security hardening). 

39 try: 

40 import sqlite3 

41 value = sqlite3.adapt(value) 

42 except ImportError: 

43 pass 

44 except sqlite3.ProgrammingError: 

45 pass 

46 # Manual emulation of SQLite parameter quoting 

47 if isinstance(value, type(True)): 

48 return str(int(value)) 

49 elif isinstance(value, (Decimal, float)): 

50 return str(value) 

51 elif isinstance(value, six.integer_types): 

52 return str(value) 

53 elif isinstance(value, six.string_types): 

54 return "'%s'" % six.text_type(value).replace("\'", "\'\'") 

55 elif value is None: 

56 return "NULL" 

57 elif isinstance(value, (bytes, bytearray, six.memoryview)): 

58 # Bytes are only allowed for BLOB fields, encoded as string 

59 # literals containing hexadecimal data and preceded by a single "X" 

60 # character: 

61 # value = b'\x01\x02' => value_hex = b'0102' => return X'0102' 

62 value = bytes(value) 

63 hex_encoder = codecs.getencoder('hex_codec') 

64 value_hex, _length = hex_encoder(value) 

65 # Use 'ascii' encoding for b'01' => '01', no need to use force_text here. 

66 return "X'%s'" % value_hex.decode('ascii') 

67 else: 

68 raise ValueError("Cannot quote parameter value %r of type %s" % (value, type(value)))

70 def _remake_table(self, model, create_field=None, delete_field=None, alter_field=None): 

71 """ 

72 Shortcut to transform a model from old_model into new_model 

73 

74 The essential steps are: 

75 1. rename the model's existing table, e.g. "app_model" to "app_model__old" 

76 2. create a table with the updated definition called "app_model" 

77 3. copy the data from the old renamed table to the new table 

78 4. delete the "app_model__old" table 

79 """ 

80 # Self-referential fields must be recreated rather than copied from 

81 # the old model to ensure their remote_field.field_name doesn't refer 

82 # to an altered field. 

83 def is_self_referential(f): 

84 return f.is_relation and f.remote_field.model is model 

85 # Work out the new fields dict / mapping 

86 body = { 

87 f.name: f.clone() if is_self_referential(f) else f 

88 for f in model._meta.local_concrete_fields 

89 } 

90 # Since mapping might mix column names and default values, 

91 # its values must be already quoted. 

92 mapping = {f.column: self.quote_name(f.column) for f in model._meta.local_concrete_fields} 

93 # This maps field names (not columns) for things like unique_together 

94 rename_mapping = {} 

95 # If any of the new or altered fields is introducing a new PK, 

96 # remove the old one 

97 restore_pk_field = None 

98 if getattr(create_field, 'primary_key', False) or ( 

99 alter_field and getattr(alter_field[1], 'primary_key', False)): 

100 for name, field in list(body.items()): 

101 if field.primary_key: 

102 field.primary_key = False 

103 restore_pk_field = field 

104 if field.auto_created: 

105 del body[name] 

106 del mapping[field.column] 

107 # Add in any created fields 

108 if create_field: 

109 body[create_field.name] = create_field 

110 # Choose a default and insert it into the copy map 

111 if not create_field.many_to_many and create_field.concrete: 

112 mapping[create_field.column] = self.quote_value( 

113 self.effective_default(create_field) 

114 ) 

115 # Add in any altered fields 

116 if alter_field: 

117 old_field, new_field = alter_field 

118 body.pop(old_field.name, None) 

119 mapping.pop(old_field.column, None) 

120 body[new_field.name] = new_field 

121 if old_field.null and not new_field.null: 

122 case_sql = "coalesce(%(col)s, %(default)s)" % { 

123 'col': self.quote_name(old_field.column), 

124 'default': self.quote_value(self.effective_default(new_field)) 

125 } 

126 mapping[new_field.column] = case_sql 

127 else: 

128 mapping[new_field.column] = self.quote_name(old_field.column) 

129 rename_mapping[old_field.name] = new_field.name 

130 # Remove any deleted fields 

131 if delete_field: 

132 del body[delete_field.name] 

133 del mapping[delete_field.column] 

134 # Remove any implicit M2M tables 

135 if delete_field.many_to_many and delete_field.remote_field.through._meta.auto_created: 

136 return self.delete_model(delete_field.remote_field.through) 

137 # Work inside a new app registry 

138 apps = Apps() 

139 

140 # Provide isolated instances of the fields to the new model body so 

141 # that the existing model's internals aren't interfered with when 

142 # the dummy model is constructed. 

143 body = copy.deepcopy(body) 

144 

145 # Work out the new value of unique_together, taking renames into 

146 # account 

147 unique_together = [ 

148 [rename_mapping.get(n, n) for n in unique] 

149 for unique in model._meta.unique_together 

150 ] 

151 

152 # Work out the new value for index_together, taking renames into 

153 # account 

154 index_together = [ 

155 [rename_mapping.get(n, n) for n in index] 

156 for index in model._meta.index_together 

157 ] 

158 

159 indexes = model._meta.indexes 

160 if delete_field: 

161 indexes = [ 

162 index for index in indexes 

163 if delete_field.name not in index.fields 

164 ] 

165 

166 # Construct a new model for the new state 

167 meta_contents = { 

168 'app_label': model._meta.app_label, 

169 'db_table': model._meta.db_table, 

170 'unique_together': unique_together, 

171 'index_together': index_together, 

172 'indexes': indexes, 

173 'apps': apps, 

174 } 

175 meta = type("Meta", tuple(), meta_contents) 

176 body['Meta'] = meta 

177 body['__module__'] = model.__module__ 

178 

179 temp_model = type(model._meta.object_name, model.__bases__, body) 

180 

181 # We need to modify model._meta.db_table, but everything explodes 

182 # if the change isn't reversed before the end of this method. This 

183 # context manager helps us avoid that situation. 

184 @contextlib.contextmanager 

185 def altered_table_name(model, temporary_table_name): 

186 original_table_name = model._meta.db_table 

187 model._meta.db_table = temporary_table_name 

188 yield 

189 model._meta.db_table = original_table_name 

190 

191 with altered_table_name(model, model._meta.db_table + "__old"): 

192 # Rename the old table to make way for the new 

193 self.alter_db_table(model, temp_model._meta.db_table, model._meta.db_table) 

194 

195 # Create a new table with the updated schema. We remove things 

196 # from the deferred SQL that match our table name, too 

197 self.deferred_sql = [x for x in self.deferred_sql if temp_model._meta.db_table not in x] 

198 self.create_model(temp_model) 

199 

200 # Copy data from the old table into the new table 

201 field_maps = list(mapping.items()) 

202 self.execute("INSERT INTO %s (%s) SELECT %s FROM %s" % ( 

203 self.quote_name(temp_model._meta.db_table), 

204 ', '.join(self.quote_name(x) for x, y in field_maps), 

205 ', '.join(y for x, y in field_maps), 

206 self.quote_name(model._meta.db_table), 

207 )) 

208 

209 # Delete the old table 

210 self.delete_model(model, handle_autom2m=False) 

211 

212 # Run deferred SQL on correct table 

213 for sql in self.deferred_sql: 

214 self.execute(sql) 

215 self.deferred_sql = [] 

216 # Fix any PK-removed field 

217 if restore_pk_field: 

218 restore_pk_field.primary_key = True 

219 

220 def delete_model(self, model, handle_autom2m=True): 

221 if handle_autom2m: 

222 super(DatabaseSchemaEditor, self).delete_model(model) 

223 else: 

224 # Delete the table (and only that) 

225 self.execute(self.sql_delete_table % { 

226 "table": self.quote_name(model._meta.db_table), 

227 })

240 def remove_field(self, model, field): 

241 """ 

242 Removes a field from a model. Usually involves deleting a column, 

243 but for M2Ms may involve deleting a table. 

244 """ 

245 # M2M fields are a special case 

246 if field.many_to_many: 

247 # For implicit M2M tables, delete the auto-created table 

248 if field.remote_field.through._meta.auto_created: 

249 self.delete_model(field.remote_field.through) 

250 # For explicit "through" M2M fields, do nothing 

251 # For everything else, remake. 

252 else: 

253 # It might not actually have a column behind it 

254 if field.db_parameters(connection=self.connection)['type'] is None: 

255 return 

256 self._remake_table(model, delete_field=field)
